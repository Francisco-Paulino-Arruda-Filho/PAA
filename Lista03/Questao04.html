string base6ParaBase2(const std::string& numeroBase6) {
                    // Mapeamento dos dígitos de base 6 para binário
                    std::unordered_map mapaBase6ParaBase2 = {
                        {'0', "000"}, {'1', "001"}, {'2', "010"}, 
                        {'3', "011"}, {'4', "100"}, {'5', "101"}
                    };
                
                    string numeroBase2;
                    for (char digito : numeroBase6) {
                        numeroBase2 += mapaBase6ParaBase2[digito];
                    }
                
                    // Remove os zeros à esquerda
                    numeroBase2.erase(0, numeroBase2.find_first_not_of('0'));
                
                    return numeroBase2.empty() ? "0" : numeroBase2;
                }

\( \text{Análise de complexidade:} \)

\( \text{Invariante de laço: O número de dígitos de base 2 é igual ao número de interações completas no laço} \)

\( \( \text{Caso base:} \)
\( \text{Se } n \text{ tamanho do vetor} = 1 \text{, então ao fim da interação esse único valor estará convertido para binário usando os valores no mapa} \)
\( \text{Caso indutivo:} \)

\(\text{Na linha do laço for, o índice é iniciado como 1, o vetor é incrementado enquanto i < n ou v[i] != x,}\)
\(\text{ se ele encontrar o valor x, a função retorna i, que o índice atual que o laço está.}\)
\(\text{Se o laço terminar sem encontrar o valor x, a função retorna -1,}\)
\(\text{Logo, independente do caso, a função retorna o índice do valor x no vetor ou -1 se não encontrar, pois graças ao for o índice é incrementado.}\)

\)
\(\text{Terminação:}\)
\(\text{Como o laço é finito, pois o laço itera encima do tamanho do vetor que representa o número e todo o número é finito.}\)
\(\text{Logo, a função termina e ao final todos os algarismos estarão convertidos}\)
