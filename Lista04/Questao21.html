\( \text{Item A:} \)

\[ T(n) = \begin{cases} 1 & \text{se } n = 1 \\ 2T (\frac{n}{4}) + 1, & \text{ se } n \geq 2 \end{cases} \]

\( \text{Se \(T(n) = O(n^{\log_b^a}) \)} \)

\( a=2, b=4, f(n)=1 \)

\( \text{Se \(1 \in O(n^{\log_4^{2-e} }) \), então \(T(n) \in \Theta(n^{\log_4^{2} }) \)} \)

\( \text{Assumindo e como \(e=1\)} \)

\( \text{Se \(1 \in O(n^{\log_4^{2-1} }) \), então \(T(n) \in \Theta(n^{\log_4^{2} }) \)} \)

\( \text{Se \(1 \in O(n^{\log_4^{1} }) \), então \(T(n) \in \Theta(n^{\log_4^{2} }) \)} \)

\( \text{Pela definição de \(O\), \(\exists c_1, n_0\) tal que \(f(n) \leq cg(n)\) } \)

\( 1 \leq cn^{\log_4^{1} } \)

\( 1 \leq cn^0 \)

\( 1 \leq c \)

\( \text{Logo, \(T(n) \in \Theta(n^\frac{1}{2} )\)} \)

\(\text{Item B:} \)

\[ T(n) = \begin{cases} 1 & \text{se } n = 1 \\ 2T (\frac{n}{4}) + \sqrt{n}, & \text{ se } n \geq 2 \end{cases} \]

\( \text{Se \(T(n) =\ O(n^{\log_b^a}) \)} \)

\( a=2, b=4, f(n)=\sqrt{n} \)

\( \text{Se \(\sqrt{n} \in O(n^{\log_4^{2} }) \), então \(T(n) \in \Theta(n^{\log_4^{2}}\log^n ) \)} \)

\( \text{Pela definição de \(O\), temos que \(\exists c_1, n_0\) tal que \(f(n) \leq cg(n)\)} \)

\( \sqrt{n} \leq cn^{\log_4^{2} } \)

\( \sqrt{n} \leq cn^{\frac{1}{2} } \)

\( n^{\frac{1}{2} } \leq cn^{\frac{1}{2} } \)

\( \text{Logo, temos que } T(n) \in \Theta(n^{\log_4^{2}}\log^n ) \)

\( \text{Logo, temos que } T(n) \in \Theta(n^{\frac{1}{2}\log^n} ) \)

\(\text{Item C:}\)

\[ T(n) = \begin{cases} 1 & \text{se } n = 1 \\ 2T (\frac{n}{4}) + n, & \text{ se } n \geq 2 \end{cases} \]

\( a=2, b=4, f(n)=n \)

\( \text{Se \(f(n) \in \Omega(n^{\log_b^{a} }) \) e \(af(\frac{n}{b}) \leq cf(n)\) , então \(T(n) \in \Theta(f(n)) \) } \)

\( \text{Se \(n \in \Omega(n^{\log_4^{2} }) \) e \(2\frac{n}{4} \leq cn\), então \(T(n) \in \Theta(n) \) } \)

\( \text{Se \(n \in \Omega(n^{\frac{1}{2} }) \) e \(\frac{n}{2} \leq cn\), então \(T(n) \in \Theta(n) \) } \)

\( \text{Logo, \(\Theta(n)\) } \)

\(\text{Item D:}\)

\[ T(n) = \begin{cases} 1 & \text{se } n = 1 \\ 2T (\frac{n}{4}) + n^2, & \text{ se } n \geq 2 \end{cases} \]

\( a=2, b=4, f(n)=n^2 \)

\( \text{Se \(f(n) \in \Omega(n^{\log_b^{a} }) \) e \(af(\frac{n}{b}) \leq cf(n)\) , então \(T(n) \in \Theta(f(n)) \) } \)

\( \text{Se \(n^2 \in \Omega(n^{\log_4^{2} }) \) e \(2\frac{n}{4} \leq cn^2\), então \(T(n) \in \Theta(n^2) \) } \)

\( \text{Se \(n^2 \in \Omega(n^{\frac{1}{2} }) \) e \(\frac{n}{2} \leq cn^2\), então \(T(n) \in \Theta(n^2) \) } \)

\( \text{Logo, \(\Theta(n^2)\) } \)
