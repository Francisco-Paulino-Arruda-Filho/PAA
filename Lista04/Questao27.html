\( \text{Utilizando qualquer método que desejar, contanto que no final deverá ter pelo menos uma} \)
\( \text{prova por substituição, por árvore de recursão e por método mestre prove as recorrências a seguir.} \)
\( \text{Assuma que para todos as alternativas T (1) = 1} \)

\( \text{Item 1:} \)

\[ T(n) = \begin{cases} 1 & \text{se } n = 1 \\ 2T (\frac{n}{2}) + n^3, & \text{ se } n \geq 2 \end{cases} \]

\( T (n) = 2T (\frac{n}{2}) + n^3. \)

\( \text{Vamos chutar: } T(n) = O(n^3) \)

\( \text{Pela definição de \(O\) temos que \(\exists n_0, c_1 \) tal que \(T(n) \leq c_1n^3 \) } \)

\( T(n) \leq 2c_1(\frac{n}{2})^3 + n^3 \)

\( 2c_1\frac{n^3}{8} + n^3 \leq c_1n^3 \)

\( \text{Basta tomar \(c_1=2\) e \(n_0 \geq 1\)} \)

\(\text{Item 2:}\)

\[ T(n) = \begin{cases} 1 & \text{se } n = 1 \\ 7T (\frac{n}{2}) + n^2, & \text{ se } n \geq 2 \end{cases} \]

\( T (n) = 7T (\frac{n}{2}) + n^2 \)

\( \text{Vamos chutar: } T(n) = O(n^3) \)

\( \text{Pela definição de \(O\) temos que \(\exists n_0, c_1 \) tal que \(T(n) \leq c_1n^3 \) } \)

\( T(n) \leq 7c_1(\frac{n}{2})^3 + n^2 \)

\( T(n) \leq 7c_1\frac{n^3}{8} + n^2 \)

\( 7c_1\frac{n^3}{8} + n^2 \leq c_1n^3 \)

\( \text{Basta tomar \(c_1=8\) e \(n_0 \geq 1\)} \)

\(\text{Item 3:}\)

\[ T(n) = \begin{cases} 1 & \text{se } n = 1 \\ T (\sqrt{n}) + 5, & \text{ se } n \geq 2 \end{cases} \]

\( T (n) = T (\sqrt{n}) + 5. \)

\( \text{Vamos chutar: } T(n) = O(n) \)

\( \text{Pela definição de \(O\) temos que \(\exists n_0, c_1 \) tal que \(T(n) \leq c_1 \) } \)

\( T(n) \leq c_1n \)

\( T(n) = T(\sqrt{n}) + 5 \)

\( T(n) \leq c_1 \sqrt{n} + 5 \)

\( c_1 \sqrt{n} + 5 \leq c_1n \)

\( \text{Basta tomar \(c_1=10\) e \(n_0 \geq 4\)} \)

\( \text{Item 4:} \)

\[ T(n) = \begin{cases} 1 & \text{se } n = 1 \\ T (n) = 3T (\frac{n}{2}) + n \log^n, & \text{ se } n \geq 2 \end{cases} \]

\( \text{Basta tomar: } T(n) = O(n^2) \)

\( \text{Pela definição de \(O\) temos que \(\exists n_0, c_1 \) tal que \(T(n) \leq c_1n \log n \) } \)

\( T(n) \leq c_1n^2 \)

\( T(n) = 3T (\frac{n}{2}) + n \log^n \)

\( T(n) \leq 3c_1(\frac{n}{2})^2 + n \log^n \)

\( T(n) \leq 3c_1\frac{n}{4} + n \log^n \)

\( 3c_1\frac{n}{4} + n \log^n \leq c_1n^2 \)

\( \text{Basta tomar \(c_1=1\) e \(n_0 \geq 4\)} \)

\( \text{Item 5:} \)

\[ T(n) = \begin{cases} 1 & \text{se } n = 1 \\ T (n) = 4T (\frac{n}{2}) + n^2 \sqrt{n}, & \text{ se } n \geq 2 \end{cases} \]

\( T (n) = 4T (\frac{n}{2}) + n^2 \sqrt{n} \)

\( \text{Chutemos: } T(n) = O(n^3) \)

\( \text{Pela definição de \(O\) temos que \(\exists n_0, c_1 \) tal que \(T(n) \leq c_1n^3 \) } \)

\( T(n) \leq c_1n^3 \)

\( T(n) = 4T (\frac{n}{2}) + n^2 \sqrt{n} \)

\( T(n) \leq 4c_1(\frac{n}{2})^3 + n^2 \sqrt{n} \)

\( T(n) \leq 4c_1\frac{n^3}{8} + n^2 \sqrt{n} \)

\( n^2 \sqrt{n} \leq n^3, \forall n \geq 1 \)

\( 4c_1\frac{n^3}{8} + n^3 \leq c_1n^3 \)

\( \text{Basta tomar \(c_1=8\) e \(n_0 \geq 1\)} \)

\( \text{Item 6:} \)

\[ T(n) = \begin{cases} 1 & \text{se } n = 1 \\ T (n) = 2T (\frac{n}{2}) + n\log^n, & \text{ se } n \geq 2 \end{cases} \]

\( \text{Chutemos: } T(n) = O(n^2) \)

\( \text{Pela definição de \(O\) temos que \(\exists n_0, c_1 \) tal que \(T(n) \leq c_1n \log n \) } \)

\( T(n) \leq c_1n^2 \)

\( T(n) = 2T (\frac{n}{2}) + n\log^n \)

\( T(n) \leq 2c_1(\frac{n}{2})^2 + n\log^n \)

\( T(n) \leq 2c_1\frac{n^2}{4} + n\log^n \)

\( n\log^n \leq n^2, \forall n \geq 1 \)

\( 2c_1\frac{n^2}{4} + n^2 \leq c_1n^2 \)

\( \text{Basta tomar \(c_1=4\) e \(n_0 \geq 1\)} \)

\( \text{Item 7:} \)

\[ T(n) = \begin{cases} 1 & \text{se } n = 1 \\ T (n) = T(n - 1) + \frac{1}{n}, & \text{ se } n \geq 2 \end{cases} \]

\( \text{Chutemos: } T(n) = O(\log n) \)

\( \text{Pela definição de \(O\) temos que \(\exists n_0, c_1 \) tal que \(T(n) \leq c_1 \log n \) } \)

\( T(n) \leq c_1\log n \)

\( T(n) = T(n - 1) + \frac{1}{n} \)

\( T(n) \leq c_1\log^{n-1} + \frac{1}{n} \)

\( c_1\log^{n-1} + \frac{1}{n} \leq c_1\log^n \)

\( \text{Basta tomar \(c_1=1\) e \(n_0 \geq 4\)} \)

\( \text{Item 8:} \)
