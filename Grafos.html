\(\text{Grafos:}\)

\(\text{Um grafo é um conjunto de vértices e arestas.}\)

\(\text{Vértices:}\) \(\text{São os pontos do grafo e são finitos}\)

\(\text{Arestas:}\) \(\text{São os segmentos que ligam os vértices e são não ordenados}\)

\(\text{Adjacênci e incidência:} \) \(\text{Dois vértices são adjacentes se estão ligados por uma aresta}\)

\(Exemplo de grafo usando mathjax \) \(\text{G = (V, E)}\)

\(\text{V = \{a, b, c, d\}}\)

\(\text{E = \{ab, ac, ad, bc, bd, cd\}}\)

\(Multigrafo:\)

\(\text{É um grafo que possui mais de uma aresta ligando dois vértices}\)

\(\text{Exemplo:}\)

\(\text{G = (V, E)}\)

\(\text{V = \{a, b, c, d\}}\)

\(\text{E = \{ab, ab, ac, ad, bc, bd, cd\}}\)

\( \text{Laço:} \)

\(\text{É uma aresta que liga um vértice a ele mesmo}\)

\(\text{Exemplo:}\)

\(\text{G = (V, E)}\)

\(\text{V = \{a, b, c, d\}}\)

\(\text{E = \{aa, ab, ac, ad, bc, bd, cd\}}\)

\( \text{Link:} \)

\(\text{É uma aresta que liga dois vértices distintos}\)

\(\text{Exemplo:}\)

\(\text{G = (V, E)}\)

\(\text{V = \{a, b, c, d\}}\)

\(\text{E = \{ab, ac, ad, bc, bd, cd\}}\)

\( \text{Tamanho do grafo} \)

\( |G|= |V| + |E| \)

\( \text{Grafos completos} \)

\(\text{Um grafo é completo se todos os vértices estão ligados por arestas}\)

\(\text{Exemplo:}\)

\(\text{G = (V, E)}\)

\(\text{V = \{a, b, c, d\}}\)

\(\text{E = \{ab, ac, ad, bc, bd, cd\}}\)

\( \text{Tamanho do grafo} \)

\( \binom{k}{n} \)

\( \text{Grau do vértice} \)

\(\text{O grau de um vértice é o número de arestas que incidem sobre ele}\)

\(\text{Exemplo:}\)

\(\text{G = (V, E)}\)

\(\text{V = \{a, b, c, d\}}\)

\(\text{E = \{ab, ac, ad, bc, bd, cd\}}\)

\(\text{Grau de a: 3}\)

\(\text{Grau de b: 3}\)

\(\text{Grau de c: 3}\)

\(\text{Grau de d: 3}\)

\( \text{Handshaking lemma:} \)

\( \text{A soma dos graus dos vértices de um grafo é igual a duas vezes o número de arestas} \)

\( \sum_{v \in V} \text{grau}(v) = 2|E| \)

\)

\( \text{Grafo complementar de grafo} \)

\(\text{O grafo complementar de um grafo é um grafo que possui os mesmos vértices do grafo original, mas as arestas são diferentes}\)

\(\text{Exemplo:}\)

\(\text{G = (V, E)}\)

\(\text{V = \{a, b, c, d\}}\)

\(\text{E = \{ab, ac, ad, bc, bd, cd\}}\)

\(\text{Grafo complementar de G:}\)

\(\text{G = (V, E)}\)

\(\text{V = \{a, b, c, d\}}\)

\(\text{E = \{ac, ad, bc, bd, cd\}}\)

\( \text{Passei em um grafo} \)

\(\text{Um passeio em um grafo é uma sequência de vértices e arestas tal que cada aresta tem como extremidades os vértices da sequência}\)

\(\text{Exemplo:}\)

\(\text{G = (V, E)}\)

\(\text{V = \{a, b, c, d\}}\)

\(\text{E = \{ab, ac, ad, bc, bd, cd\}}\)

\(\text{Passeio: a, b, c, d}\)

\( \text{Grafo direcionado} \)

\(\text{Um grafo direcionado é um grafo cujas arestas são orientadas}\)

\(\text{Exemplo:}\)

\(\text{G = (V, E)}\)

\(\text{V = \{a, b, c, d\}}\)

\(\text{E = \{ab, ac, ad, bc, bd, cd\}}\)

\(\text{Grafo direcionado:}\)

\(\text{G = (V, E)}\)

\(\text{V = \{a, b, c, d\}}\)

\(\text{E = \{(a, b), (a, c), (a, d), (b, c), (b, d), (c, d)\}}\)

\( \text{Grafo de caminhos} \)

\(\text{Um grafo de caminhos é um grafo cujas arestas são orientadas e não formam ciclos}\)

\(\text{Exemplo:}\)

\(\text{G = (V, E)}\)

\(\text{V = \{a, b, c, d\}}\)

\(\text{E = \{(a, b), (b, c), (c, d)\}}\)

\( \text{Grafo de ciclos} \)

\(\text{Um grafo de ciclos é um grafo cujas arestas são orientadas e formam ciclos}\)

\(\text{Exemplo:}\)

\(\text{G = (V, E)}\)

\(\text{V = \{a, b, c, d\}}\)

\(\text{E = \{(a, b), (b, c), (c, d), (d, a)\}}\)

\( \text{Conexidade} \)

\(\text{Um grafo é conexo se existe um passeio entre quaisquer dois vértices}\)

\(\text{Exemplo:}\)

\(\text{G = (V, E)}\)

\(\text{V = \{a, b, c, d\}}\)

\(\text{E = \{ab, ac, ad, bc, bd, cd\}}\)

\( \text{Grafo ponderado} \)

\(\text{Um grafo ponderado é um grafo cujas arestas possuem um peso}\)

\(\text{Exemplo:}\)

\(\text{G = (V, E)}\)

\(\text{V = \{a, b, c, d\}}\)

\(\text{E = \{ab, ac, ad, bc, bd, cd\}}\)

\(\text{Peso de ab: 2}\)

\(\text{Peso de ac: 3}\)

\(\text{Peso de ad: 4}\)

\(\text{Peso de bc: 5}\)

\(\text{Peso de bd: 6}\)

\(\text{Peso de cd: 7}\)
