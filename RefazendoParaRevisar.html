\(\text{Questão 2} \)

\[ f(n) = 3n^2+7n-8 = O(n^2) \]

\(\text{Enunciado}\)

\(\text{Mostre que a função } f(n) = 3n^2+7n-8 = O(n^2)\)

\[f(n)=3n^2+7n-8 \]

\[g(n)=n^2\]

\(\text{Pela definição de O, temos que \(f(n) = O(g(n))\) se existirem constantes positivas \(c\) e \(n_0\) tais que \(0 \leq f(n) \leq cg(n)\) para todo \(n \geq n_0\).}\)

\(f(n) \leq cg(n) \)

\(3n^2+7n-8 \leq cn^2 \)

\(7n \leq 7n^2, \forall n \geq 1 \)

\(-8 \leq n^2, \forall \geq 1 \)

\(3n^2+7n-8 \leq 3n^2+7n^2+n^2, \forall n \geq 1 \)

\(3n^2+7n-8 \leq 11n^2\)

\( 11n^2 \leq n^2 \)

\(\text{Basta tomar } c=11 \text{ e } n_0=1 \)

\(\text{Questão 3} \)

\(\text{Enunciado}\)

\(\text{Mostre que \(100n = O(2n)\), que \(n + 100 = O(n)\), e que \(100n = O(2n^2 + n)\).} \)

\(\text{Primeira parte:} \)

\(\text{Mostre que } 100n = O(2n) \)

\[f(n)=100n \]

\[g(n)=2n\]

\(\text{Pela definição de O, temos que \(f(n) = O(g(n))\) se existirem constantes positivas \(c\) e \(n_0\) tais que \(0 \leq f(n) \leq cg(n)\) para todo \(n \geq n_0\).}\)

\(f(n) \leq cg(n) \)

\(100n \leq c2n \)

\(100 \leq 2, \forall n \geq 1 \)

\(\text{Basta tomar } c=50 \text{ e } n_0=1 \)

\(\text{Segunda parte:} \)

\(\text{Mostre que } 100n = O(2n^2+n) \)

\[f(n)=100n \]

\[g(n)=2n^2+n \]

\(\text{Pela definição de O, temos que \(f(n) = O(g(n))\) se existirem constantes positivas \(c\) e \(n_0\) tais que \(0 \leq f(n) \leq cg(n)\) para todo \(n \geq n_0\).}\)

\(f(n) \leq cg(n) \)

\(100n \leq c(2n^2+n) \)

\(100n \leq 100n^2, \forall n \geq 1 \)

\(100n^2 \leq c(2n^2+n) \)

\(\text{Basta tomar \(c=50\) e \(n_0=1\) } \)

\(\text{Questão 6} \)

\[ \frac{1}{2}n(n - 1) = O(n^2). \]

\(\text{Enunciado}\)

\(\text{Mostre que a função } f(n) = \frac{1}{2}n(n-1) = O(n^2)\)

\[f(n)=\frac{1}{2}n(n-1) \]

\[g(n)=n^2\]

\(\text{Pela definição de O, temos que \(f(n) = O(g(n))\) se existirem constantes positivas \(c\) e \(n_0\) tais que \(0 \leq f(n) \leq cg(n)\) para todo \(n \geq n_0\).}\)

\(f(n) \leq cg(n) \)

\(\frac{1}{2}n(n-1) \leq cn^2 \)

\(\frac{1}{2}(n^2-n) \leq cn^2 \)

\(\text{\(n^2-n \leq cn^2,\forall n \geq 1\)} \)

\(\text{Basta tomar } c=\frac{1}{2} \text{ e } n_0=1 \)
