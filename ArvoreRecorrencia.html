\[ T(n) = \begin{cases} 1 & \text{se } n \leq 3 \\ 3T(\floor \frac{n}{4} \rfloor)+cn^2 , & \text{ se } n \geq 4 \end{cases} \]

\[ \begin{array}{|c|c|c|} \hline \text{Nível \(i\)} & \text{Tamanho da entrada} & \text{Nós} & \text{Tempo nó} \\ \hline 1 & n & 1 & n \\ 2 & \frac{n}{4} & 4 & \frac{n}{4} \\ 3 & \frac{n}{16} & 16 & \frac{n}{9} \\ 4 & \frac{n}{16} & 16 & \frac{n}{16} \\ x & 1 & 4^x & 1 \\ \hline \end{array} \]

\[ \frac{n}{4^x} = 1 \Rightarrow x = \log_{4}(n) \]

\[ \begin{array}{|c|c|c|} \hline \text{Nível \(i\)} & \text{Tamanho da entrada} & \text{Nós} & \text{Tempo nó} \\ \hline 1 & n & 1 & n \\ 2 & \frac{n}{4} & 4 & \frac{n}{4} \\ 3 & \frac{n}{16} & 16 & \frac{n}{9} \\ 4 & \frac{n}{64} & 64 & \frac{n}{64} \\ \log^n_4 & \frac{n}{4^{\log^n_4}} & 4^{\log^n_4} & \frac{n}{4^{\log^n_4}} \\ \hline \end{array} \]

\[ \begin{array}{|c|c|c|} \hline \text{Nível \(i\)} & \text{Tamanho da entrada} & \text{Nós} & \text{Tempo nó} \\ \hline 1 & n & 1 & n \\ 2 & \frac{n}{4} & 4 & \frac{n}{4} \\ 3 & \frac{n}{16} & 16 & \frac{n}{9} \\ 4 & \frac{n}{64} & 64 & \frac{n}{64} \\ \log^n_4 & 1 & n & 1 \\ \hline \end{array} \]

\( \text{Termo geral} \)

\[ 3^i \cdot \left(\frac{n}{4^i}\right) \]

\( \text{Soma dos termos} \)

\[ \sum_{i=0}^{\log^n_4} 3^i \cdot \left(\frac{n}{4^i}\right) \]

\( \text{Simplificando} \)

\[ \sum_{i=0}^{\log^n_4} n \cdot \left(\frac{3}{4}\right)^i \]

\( \text{Soma de PG infinita:} \)

\[ S_{\infty} = \frac{a}{1 - q} \]

\[ S_{\infty} =\frac{n}{1 - \frac{3}{4}} \]

\[ S_{\infty} =\frac{n}{\frac{1}{4}} \]

\[ S_{\infty} =4n \]

\[ \sum_{i=0}^{\log^n_4} cn^2 \]

\( = \log^n_4 \cdot cn^2 \)

\[ T(n) \leq 4n + \log^n_4 \cdot cn^2\ \]

\[ O(n^2) \]
