\( \text{Questão 01: Utilize o método da substituição para calcular a complexidade de tempo de pior caso do algoritmo abaixo.} \) 

            
                Algoritmo(N):
                Se N == 0 entao
                  Retorne 0
                Senao
                  soma = 0
                  Para j = 1, ..., N faça
                    soma = soma + j
                  Retorne soma + Algoritmo(N/2)

\[ T(n) = \begin{cases} 1 & \text{se } n = 1 \\ T(N)=N+T(\frac{N}{2}) , & \text{ se } n \geq 2 \end{cases} \]

\( \text{Vamos chutar que \(T(n) = O(n)\).} \)

\( \text{Pela definição de \(O\), temos que: \(\exists c_1, n_0\) tal que \(T(n) \leq c_1n\) para \(n \geq n_0\).} \)

\( T(n) = T(\frac{n}{2}) + O(n) \)

\( T(N)=N+T(\frac{N}{2}) \)

\( T(N) \leq c_1\frac{N}{2} + N \)

\( c_1\frac{N}{2} + N \leq c_1N \)

\( \text{Basta tomar \(c_1 = 2\) e \(n_0 \geq 2\).} \)

\( \text{Questão 02:} \)

\( \text{A: \(T(n)=T(\frac{n}{2})+15\), onde \(T(1)=1\).} \)

\[ T(n) = \begin{cases} 1 & \text{se } n = 1 \\ T(\frac{n}{2})+15, & \text{ se } n \geq 2 \end{cases} \]

\( \text{Vamos chutar que \(T(n) = O(n)\).} \)

\( \text{Pela definição de \(O\), temos que: \(\exists c_1, n_0\) tal que \(T(n) \leq c_1n\) para \(n \geq n_0\).} \)

\( T(n)=T(\frac{n}{2})+15 \)

\( T(n) \leq c_1\frac{n}{2}+15 \)

\( c_1\frac{n}{2}+15 \leq c_1n \)

\( \text{Basta tomar \(c_1 = 2\) e \(n_0 \geq 15\).} \)

\( \text{Item B: \(F(n)=2F(\frac{n}{2})+n, \text{ onde } F(1)=1.\)} \)

\[ F(n) = \begin{cases} 1 & \text{se } n = 1 \\ 2F(\frac{n}{2})+n, & \text{ se } n \geq 2 \end{cases} \]

\( \text{Vamos chutar que \(F(n) = O(n\log n)\).} \)

\( \text{Pela definição de \(O\), temos que: \(\exists c_1, n_0\) tal que \(F(n) \leq c_1n\log n\) para \(n \geq n_0\).} \)

\( F(n)=2F(\frac{n}{2})+n \)

\( F(n) \leq 2c_1(\frac{n}{2})\log(\frac{n}{2})+n \)

\( 2c_1(\frac{n}{2})\log(\frac{n}{2})+n \leq c_1n\log n \)

\( c_1n\log(\frac{n}{2})+n \leq c_1n\log n \)

\( \text{Basta tomar \(c_1 = 2\) e \(n_0 \geq 2\).} \)

\( \text{C: \(G(n)=2G(\sqrt{n})\), onde \(G(1)=1.\)} \)

\[ G(n) = \begin{cases} 1 & \text{se } n = 1 \\ 2G(\sqrt{n}), & \text{ se } n \geq 2 \end{cases} \]

\( \text{Vamos chutar que \(G(n) = O(n)\).} \)

\( \text{Pela definição de \(O\), temos que: \(\exists c_1, n_0\) tal que \(G(n) \leq c_1n\) para \(n \geq n_0\).} \)

\( G(n) \leq 2c_1n \)

\( 2c_1\sqrt{n} \leq c_1n \)

\( \text{Basta tomar \(c_1 = 2\) e \(n_0 \geq 4\).} \)

\( \text{Questão 03: Usando o método da substituição, prove que a solução para a relação de recorrência \(T(n)=2T(\lfloor \frac{n}{2} \rfloor +17)+n\) é \(O(n\log^n)\).} \)

\[ T(n) = \begin{cases} 1 & \text{se } n = 1 \\ 2T(\lfloor \frac{n}{2} \rfloor +17)+n, & \text{ se } n \geq 2 \end{cases} \]

\( \text{Queremos provar que \(T(n) \leq cn\log n\).} \)
\( \text{\(\forall n \geq n_0\) tal que \(c, n_0 \in \mathbb{R}_+\ \) } \)
\( \text{Vou usar indução em \(n\).} \)

\( \text{Hipótese indutiva: } \forall n_0 \leq k < n, T(k) \leq c(k-34)\log^{k-34} \)

\( \text{Passo indutivo: Quero provar que \(T(n) \leq c(n-34)\log^{n-34}\).} \)

\( T(n) = 2T(\lfloor \frac{n}{2} \rfloor +17)+n \)

\( T(n) \leq 2c(\lfloor \frac{n}{2} \rfloor +17-34)\log^{\lfloor \frac{n}{2} \rfloor +17-34}+n \)

\( = 2c(\frac{n}{2}-17)\log^{\frac{n}{2}-17}+n \)

\( = 2c(\frac{n-34}{2})\log^{\frac{n-34}{2}}+n \)

\( = c(n-34)\log^{\frac{n-34}{2}}+n \)

\( = c(n-34)(\log^{n-34} - \log^{2})+n \)

\( = c(n-34)(\log^{n-34} - 1)+n \)

\( = c(n-34)\log^{n-34} - c(n-34)+n \)

\( c(n-34)\log^{n-34} - c(n-34)+n \leq c(n-34)\log^{n-34} \)

\( - c(n-34)+n \leq 0 \)

\( \text{Basta tomar \(c = 2\) e \(n_0 \geq 68\).} \)

\( \text{Questão 04:} \)

\(\text{Usando o Teorema Mestre da Seção 4.5 do livro, você pode mostrar que a solução}\)

\(\text{com que uma prova de substituição funcione}\)

\( 4c(\frac{n}{3})\log^{4}_{3} + n \leq cn\log^{4}_{3} \)

\( 4c(\frac{n}{3}) \geq cn \)

\( 4cn \geq 3cn, \forall n \geq 1 \)

\(\text{Logo, a prova falha}\)

\(\text{Suponha que \(T(n) = O(n^2)\)}\)

\( \text{Pela definição de O, \(\exists c_1, n_0 \) tal que } T(n) \leq cg(n) \)

\( T(n) \leq cn^2 \)

\( T(n) \leq 4c(\frac{n}{9}) + n \)

\( 4c(\frac{n^2}{9}) + n \leq cn^2 \)

\( n \leq n^2, \forall n \geq 1 \)

\( 4*2(\frac{n^2}{9}) + n^2 \leq 2n^2 \)

\( 8c(\frac{n^2}{9}) \leq n^2, \forall n \geq 1 \)

\( \text{Logo, é verdade que \(T(n) = O(n^2)\)} \)
