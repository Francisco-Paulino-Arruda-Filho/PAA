\(\text{Questão 1} \) \( \text{Projete um algoritmo para calcular \(\lfloor \sqrt{n} \rfloor\) utilizando apenas comandos de seleção, repetição e atribuição,} \)
\( \text{e as quatro operações aritméticas básicas. Obs.: Como apenas a parte inteira do valor raiz quadrada é solicitado,} \)
\( \text{use um algoritmo iterativo simples que testa todos os valores inteiros dentro de uma faixa de valores possíveis.} \) 

                    
                        int raiz_quadrada(int n) {
                            int i = 0;
                            while (i * i <= n) {
                                i++;
                            }
                            return i - 1;
                        }

\(\text{Questão 3} \)

\(\text{Prove as seguintes afirmações:} \)

  • \[ (n+2)^2 = O(n^2) \]

    \[ f(n) = (n+2)^2 \]

    \[ g(n) = n^2 \]

    \(\text{Pela definição de O \(\exists n_0, c\) tal que \(f(n) \leq cn^2 \) } \)

    \( (n+2)^2 \leq cn^2 \)

    \( n^2 + 4n + 4 \leq cn^2 \)

    \( 4n \leq 4n^2, \forall n \geq 1 \)

    \(4 \leq 4n^2, \forall n \geq 1 \)

    \( n^2 + 4n + 4 \leq n^2 + 4n^2 + 4n^2, \forall n \geq 1 \)

    \( n^2 + 4n^2 + 4n^2 \leq cn^2 \)

    \( 9n^2 \leq cn^2 \)

    \(\text{Basta tomar \(c=9\) e \(n_0=1\) } \)

  • \[ n^2 -3n -18 = \Omega(n^2) \]

    \[ f(n) = n^2 -3n -18 \]

    \[ g(n) = n^2 \]

    \(\text{Pela definição de \(\Omega \) \(\exists n_0, c\) tal que \(f(n) \geq cn^2 \) } \)

    \( n^2 -3n -18 \geq cn^2 \)

    \( \text{Basta assumir } c=\frac{1}{2} \)

    \( n^2 -3n -18 \geq \frac{1}{2}n^2 \)

    \( \frac{1}{2}n^2 -3n -18 \geq 0 \)

    \( n^2 -6n -36 \geq 0 \)

    \( \text{Resolvendo a inequação} \)

    \( \frac{6 \pm \sqrt{6^2 - 4 \cdot 1 \cdot (-36)}}{2 \cdot 1} \)

    \( \frac{6 \pm \sqrt{36 + 144}}{2} \)

    \( \frac{6 \pm \sqrt{180}}{2} \)

    \( \frac{6 \pm 6\sqrt{5}}{2} \)

    \( 3 \pm 3\sqrt{5} \)

    \( 3 - 3\sqrt{5} \approx -6.7 \)

    \( 3 + 3\sqrt{5} \approx 12.7 \)

    \( \text{Logo, para } n_0 \geq 13 \text{ a inequação é verdadeira.} \)

  • \[ n^3 -3n^2 - n = \Omega(n^3) \]

    \[ f(n) = n^3 -3n^2 - n \]

    \[ g(n) = n^3 \]

    \(\text{Pela definição de \(\Omega \) \(\exists n_0, c\) tal que \(f(n) \geq cn^3 \) } \)

    \( n^3 -3n^2 - n \geq cn^3 \)

    \(\text{Basta tomar } c=\frac{1}{2} \)

    \( n^3 -3n^2 - n \geq \frac{1}{2}n^3 \)

    \( \frac{1}{2}n^3 -3n^2 - n \geq 0 \)

    \( \frac{1}{2}n^2 - 3n - 1 \geq 0 \)

    \( n^2 - 6n - 2 \geq 0 \)

    \(\text{Resolvendo a inequação} \)

    \( \frac{6 \pm \sqrt{6^2 - 4 \cdot 1 \cdot (-2)}}{2 \cdot 1} \)

    \( \frac{6 \pm \sqrt{36 + 8}}{2} \)

    \( \frac{6 \pm \sqrt{44}}{2} \)

    \( \frac{6 \pm 2\sqrt{11}}{2} \)

    \( 3 \pm \sqrt{11} \)

    \( 3 - \sqrt{11} \approx 0.08 \)

    \( 3 + \sqrt{11} \approx 6.92 \)

    \( \text{Logo, para } n_0 \geq 7 \text{ a inequação é verdadeira.} \)

  • \[ \lg^{100n^3+200n300} = O(\lg n) \]

    \[ f(n) = \lg^{100n^3+200n300} \]

    \[ g(n) = \lg n \]

    \(\text{Pela definição de O temos que \(\exists n_0, c\) tal qual \(\lg^{(100n^3+200n+300)^2} \leq c\lg^n \) } \)

    \( \lg^{(100n^3+200n+300)^2} \leq c\lg^n \)

    \( 2\lg^{100n^3+200n+300} \leq c\lg^n \)

    \(\text{Basta tomar} c=8 \)

    \( 2\lg^{100n^3+200n+300} \leq 8\lg^n \)

    \( \lg^{100n^3+200n+300} \leq 4\lg^n \)

    \( \lg^{100n^3+200n+300} \leq \lg^{n^4} \)

    \( 100n^3+200n+300 \leq n^4 \)

    \( 300 \leq n^4 - 100n^3 - 200n \)

    \( 300 \leq n(n^3 - 100n^2 - 200) \)

    \( 300 \leq n(n^2(n - 100) - 200) \)

    \( \text{Para basta tomar} n=105 \)

    \( 300 \leq 105(11025(5) - 200) \)

    \( 300 \leq 105(55125 - 200) \)

    \( 300 \leq 105(54925) \)

    \( 300 \leq 5767125 \)

    \( \text{Logo, para } n_0 \geq 105 \text{ a inequação é verdadeira.} \)
