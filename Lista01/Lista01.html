\(\text{Problema:} \)

\(\text{Um problema é uma questão ou situação que precisa ser resolvida.} \)

\(\text{Algoritmo:} \)

\(\text{Um algoritmo é um conjunto de instruções finitas e bem definidas que, quando executadas, resolvem um problema.} \)

\(\text{Modelo computacional RAM:} \)

\(\text{O modelo computacional RAM (Random Access Machine) é uma máquina abstrata que possui uma memória infinita e instruções simples.} \)

\(\text{Complexidade de tempo:} \)

\(\text{A complexidade de tempo de um algoritmo é a quantidade de tempo que ele leva para executar em função do tamanho da entrada.} \)

                        soma_binarios(array A, array B, int n)
                            array C[n+1]
                            int carry = 0
                            for i = 0 to n do
                                if(A[i] + B[i] + carry == 0) then
                                    C[i] = 0
                                    carry = 0
                                else if(A[i] + B[i] + carry == 1) then
                                    C[i] = 1
                                    carry = 0
                                else if(A[i] + B[i] + carry == 2) then
                                    C[i] = 0
                                    carry = 1
                                else
                                    C[i] = 1
                                    carry = 1
                            C[n] = carry
                            return C

                    eh_quadrado_magico(int n, array M)
                        int soma = 0
                        for i = 0 to n-1 do
                            soma = soma + M[0][i]
                        int soma_linha = 0
                        for i = 0 to n-1 do
                            soma_linha = 0
                            for j = 0 to n-1 do
                                soma_linha = soma_linha + M[i][j]
                            if(soma_linha != soma) then
                                return false
                        int soma_coluna = 0
                        for i = 0 to n-1 do
                            soma_coluna = 0
                            for j = 0 to n-1 do
                                soma_coluna = soma_coluna + M[j][i]
                            if(soma_coluna != soma) then
                                return false
                        int soma_diagonal1 = 0
                        for i = 0 to n-1 do
                            soma_diagonal1 = soma_diagonal1 + M[i][i]
                        if(soma_diagonal1 != soma) then
                            return false
                        int soma_diagonal2 = 0
                        for i = 0 to n-1 do
                            soma_diagonal2 = soma_diagonal2 + M[i][n-1-i]
                        if(soma_diagonal2 != soma) then
                            return false
                        return true
